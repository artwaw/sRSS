65 void MainWindow::zeroconf() {
66  QDir path(QStandardPaths::writableLocation(QStandardPaths::AppDataLocation));
67  if (!path.exists()) {
68  path.mkpath(QStandardPaths::writableLocation(QStandardPaths::AppDataLocation));
69  }
70  settings.setValue("dbfile",QStandardPaths::writableLocation(QStandardPaths::AppDataLocation)+"/sRSS.srss");
71  settings.setValue("cache",QStandardPaths::writableLocation(QStandardPaths::CacheLocation));
72  settings.setValue("envelopes",true);
73  settings.setValue("downloads",QStandardPaths::writableLocation(QStandardPaths::DownloadLocation));
74  settings.setValue("askdownload",false);
75  settings.setValue("autocheck",true);
76  settings.setValue("autointerval",15);
77  settings.setValue("honor",true);
78  saveSettings();
79  if (!initDB()) {
80  criticalMsg(tr("Error opening file"),tr("There was an error creating data file using default filename and path. Please go to settings and verify the data file location."));
81  return;
82  }
83  QSqlQuery query;
84  query.exec("create table sIndex(id integer primary key, rgroup integer, title text not null, link text not null unique, description text not null, lang text, copyright text, mandir text, webmaster text, pubdate text, lastbuild text, category text, generator text, docs text, ttl integer, imgurl text, imgwidth integer, imgheight integer, skiphrs text, skipdays text);");
85  if (query.lastError().type()!=QSqlError::NoError) {
86  fail(1,tr("Error while data file init"),tr("Something went wrong while preparing the data file. Please forward the below info to srss@trollnet.com.pl: sIndex - ")+query.lastError().text());
87  }
88  query.exec("insert into sIndex(title,link,description) values('Unread','unread','All unread items');");
89  query.exec("insert into sIndex(title,link,description) values('Bookmarked','bookmarked','All bookmarked items');");
90  query.exec("create table items(id integer primary key, origin integer not null, title text, link text, description text, author text, category text, comments text, guid text, enclink text, enclen integer, enctype text, pubdate text, star bool, read bool);");
91  if (query.lastError().type()!=QSqlError::NoError) {

148 void MainWindow::addChannel() {
149  class addChannel addDialog;
150  if (addDialog.exec()==QDialog::Rejected) { return;}
151  ChannelParser::channelInfo info=addDialog.getChannelInfo();
152  QVector<ChannelParser::itemInfo> chinfo=addDialog.getChannelItems();
153  QSqlRecord rec = channels->record();
154  rec.setGenerated("id",false);
155  rec.setValue("title",info.title);
156  rec.setValue("link",info.link.toString());
157  rec.setValue("description",info.description);
158  rec.setValue("lang",info.language.name());
159  rec.setValue("copyright",info.copyright);
160  rec.setValue("mandir",info.managingDirector);
161  rec.setValue("webmaster",info.webMaster);
162  rec.setValue("pubdate",info.pubDate.toString(Qt::ISODate));
163  rec.setValue("lastbuild",info.lastBuildDate.toString(Qt::ISODate));
164  rec.setValue("category",info.category);
165  rec.setValue("generator",info.generator);
166  rec.setValue("docs",info.docs.toString());
167  rec.setValue("ttl",info.ttl);
168  channels->insertRecord(-1,rec);
169  int id = channels->record(channels->rowCount()-1).value(0).toInt();
170  channels->select();
171  rec = items->record();
172  if (chinfo.isEmpty()) { return; }
173  for (auto x=0;x<chinfo.size();++x) {
174  rec.setValue("origin",id);
175  rec.setValue("title",chinfo.at(x).title);
176  rec.setValue("link",chinfo.at(x).link.toString());
177  rec.setValue("description",chinfo.at(x).description);
178  rec.setValue("author",chinfo.at(x).author.toString());
179  rec.setValue("comments",chinfo.at(x).comments.toString());
180  rec.setValue("guid",chinfo.at(x).guid);
181  rec.setValue("encurl",chinfo.at(x).enclosure.url.toString());
182  rec.setValue("enclen",chinfo.at(x).enclosure.length);
183  rec.setValue("enctype",chinfo.at(x).enclosure.type);
184  rec.setValue("pundate",chinfo.at(x).pubDate.toString(Qt::ISODate));
185  rec.setValue("star",false);
186  rec.setValue("read",false);
187  items->insertRecord(-1,rec);
188  rec.clearValues();
189  }
190  items->select();
191 }
