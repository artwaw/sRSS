79 qint64 optiondlg::getDirSize(QString dir) {
80  QDir folder(dir);
81  qint64 sum=0;
82  QFileInfoList flist=folder.entryInfoList(QDir::NoDot|QDir::NoDotDot|QDir::NoSymLinks);
83  for (auto x=0;x<flist.size();++x) {
84  if (flist.at(x).isDir()) {
85  sum=getDirSize(flist.at(x).dir().path());
86  } else {
87  sum=sum+flist.at(x).size();
88  }
89  }
90  return sum;
91 }
